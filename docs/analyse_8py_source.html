212  if len(arl1) != len(arl2):
213  raise ValueError('AtomsRow selections are not of the same length!')
214  rmsd_list = np.zeros(len(arl1))
215 
216  # Compare through every molecule pair.
217  for i, (a1, a2) in enumerate(zip(arl1, arl2)):
218  mol1 = a1.toatoms()
219  mol2 = a2.toatoms()
220  if len(mol1) != len(mol2):
221  raise ValueError(f'Atoms objects have different lengths! arl1 has length {len(arl1)}, while arl2 has length {len(arl2)}.')
222 
223  # Rotate/translate mol2 to get the least RMSD.
224  minimize_rotation_and_translation(mol1, mol2)
225 
226  rmsd = 0.0
227  num_H = 0
228 
229  for j, (atom1, atom2) in enumerate(zip(mol1, mol2)):
230  if atom1.symbol != atom2.symbol:
231  raise ValueError('Molecules have different atomic mapping!')
232 
233  if ignore_H and atom1.symbol == 'H':
234  num_H += 1
235  continue
236 
237  pos_diff = np.linalg.norm(atom1.position - atom2.position)**2
238  rmsd += pos_diff
239 
240  if ignore_H:
241  len_no_H = len(mol1) - num_H
242  rmsd *= 1/len_no_H
243  else:
244  rmsd *= 1/len(mol1)
245

446  ## @brief Calculate the mean differences in bond lengths between the molecules in two DFdb sets.
447  #
448  # The molecules in each of the sets should be calculated from the same structure
449  # files to ensure identical atomic mapping.
450  #
451  # Separates out bond length differences by bond type, providing alist of bond length
452  # differences for each unique bond type encountered in the molecules.
453  #
454  #
455  # @param arl1 ***list*** <br>
456  # List of AtomsRows containing a set of AtomsRows with identical chemical formulae to those in ari2.
457  # @param arl2 ***list*** <br>
458  # List of AtomsRows containing a set of AtomsRows with identical chemical formulae to those in ari1.
459  #
460  # @return
461  # A list of bond length differences with a value for each pair of rows in arl1
462  # and arl2, alongside a list of corresponding bond type descriptors (eg. 'C-H').
463  #
464  #
465  # @protected
466  def _compare_bond_lengths(self, arl1, arl2):
467  # Sanity check.
468  if len(arl1) != len(arl2):
469  raise ValueError('AtomsRow selections are not of the same length!')
470 
471  # Set up output dict.
472  mean_diffs_by_btype = {}
473 
474  # Compare through every molecule pair.
475  for i, (a1, a2) in enumerate(zip(arl1, arl2)):
476  mol1 = a1.toatoms()
477  mol2 = a2.toatoms()
478  if len(mol1) != len(mol2):
479  raise ValueError(f'Atoms objects have different lengths! arl1 has length {len(arl1)}, while arl2 has length {len(arl2)}.')
480 
481  # Set up analysis objects and get atom connections.
482  ana1 = Analysis(mol1)
483  ana2 = Analysis(mol2)
484  bonds1 = ana1.unique_bonds[0]
485  bonds2 = ana2.unique_bonds[0]
486 
487  # Loop through connections, get bond lengths and calculate differences.
488  bond_diffs = []
489  for i, (bondedto1, bondedto2) in enumerate(zip(bonds1, bonds2)):
490  if bondedto1 != bondedto2:
491  raise RuntimeError('Molecules have different connectivity matrices.')
492  if bondedto1 == []:
493  continue
494  for j1, j2 in zip(bondedto1, bondedto2):
495  atom_type_i1 = mol1[i].symbol
496  atom_type_j1 = mol1[j1].symbol
497  atom_type_i2 = mol2[i].symbol
498  atom_type_j2 = mol2[j2].symbol
499  # Another sanity check.
500  if atom_type_i1 != atom_type_i2 or atom_type_j1 != atom_type_j2:
501  raise RuntimeError('Molecules have different atomic indexing.')
502  # Deal with same-type interactions, eg. C-C bonds.
503  if atom_type_i1 == atom_type_j1:
504  atom_type_j1 = f'{atom_type_i1}*'
505 
506  bl1 = ana1.get_bond_value(0, (i, j1))
507  bl2 = ana2.get_bond_value(0, (i, j2))
508  bl_diff = bl2-bl1
509 
510  # Using a set so elements can be in reverse order.
511  # Using a frozenset so it is immutable and can be used as a key for a dict.
512  btype = frozenset((atom_type_i1, atom_type_j1))
513 
514  bond_diffs.append([btype, bl_diff])
515 
516  # Free up memory.
517  ana1.clear_cache()
518  ana2.clear_cache()
519 
520  # Assemble list of unique bond types.
521  unique_btypes = []
522  for btype, _ in bond_diffs:
523  if btype not in unique_btypes:
524  unique_btypes.append(btype)
525 
526  # Average over each bond type.
527  num_btypes = len(unique_btypes)
528  bl_diffs_by_btype = np.zeros(len(unique_btypes))
529  btype_counters = [0 for _ in range(num_btypes)]
530  for i, unique_btype in enumerate(unique_btypes):
531  for btype, b_diff in bond_diffs:
532  if unique_btype == btype:
533  btype_counters[i] += 1
534  bl_diffs_by_btype[i] += b_diff
535 
536  # Calculate mean differences in bond lengths of each type within this molecule.
537  mean_diffs = np.array([bl_diffs_by_btype[i] / btype_counters[i] for i in range(num_btypes)])
